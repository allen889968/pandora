    });
});

// 示例361
$("#demo361").click(function () {
    pandora.dialog({
        title: "关闭对话框前执行函数",
        content: "这里是一些可保存的表单数据，如果用户填写了信息但没有保存，则关闭窗口前会有提醒用户保存信息。",
        beforeunload: function () {
            var that = this;
            
            // 判断条件(关闭前提醒则为true)
            var bool = true;
            
            if(bool){
                $.confirm("关闭弹窗会导致已经填写的数据遗失，确定要关闭弹窗？", function () {
                    delete that.config.beforeunload;
                    that.close();
                });
                return false;
            }
        },
        ok: true,
        cancel: true
        
    });
});

// 示例：ajax 调用示例

// 示例411
pandora.dialog({
    initialize: function () {
        var that = this;
        that.loading();
        $.ajax({
            type: "GET",
            url: "ajax.html",
            dataType: "html",
            success: function (data) {
                
                // 模拟ajax阻塞两秒钟
                var timed = (new Date()).getTime();
                var bool = true;
                while (bool) {
                    if (timed + 2000 < (new Date()).getTime()) {
                        bool = false;
                    }
                }
                
                // 显示ajax数据
                that.content($(data).find("#data").html());
            }
        });
    }
});
