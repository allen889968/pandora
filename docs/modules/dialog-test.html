// 示例141
// 方法一 遮罩当前页，在iframe中时，不能遮盖iframe的父页面
pandora.loading("正在保存中...",function(){
    var that = this;
    
    //自定义操作方法，如：
    $(".overlay").click(function(){
        that.close();   // 关闭弹窗
    })
});

// 方法二 无论有无iframe页面，都会遮盖完整浏览器窗口
top.pandora.loading("正在保存中...",function(){
    var that = this;
    
    // 自定义方法中调用 that.close(); 例如：
    $(top.document).find(".overlay").click(function(){
        that.close();   // 关闭弹窗
    })
    
});
