// 示例142 iframe子页面调用遮罩住父页面
$("#demo142").click(function(){
    top.pandora.loading("正在保存中...",function(){
        var that = this;
        
        //自定义操作方法
        $(top.document).find(".overlay").click(function(){
            
            //关闭弹窗
            that.close();
        })
        
    });
})
