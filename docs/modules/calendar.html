日历排版布局

日历或作为产品价格表，或作为日期选择框，但总体结构或实现，很多地方都是相似甚至通用，所以我们考虑，一套结构多处使用，可以用作大型的产品列表，可以用于小处的日期选择。根据模块开发思想，js的实现分为两步，第一步是实现日历功能，第二步填充所需的数据，通过选择主题或版式实现大/小日历。

// 标准模式，默认的参数配置：
// 调用方式 pandora.calendar({配置修改下面所提供的参数即可});

pandora.calendar.defaults = {
        date: new Date(),
        target: $("body"),
        trigger: ".calendar", //对话框触发点/触发事件对象
        triggerEvent: "click", // 对话框触事件
        triggerClass: "calendar",// 对话样式
        warpClass: "ui-calendar", // 日历最外层样式名称
        selectDateCallback: null, // 选择日期后回调函数
        autoRender: false,// 是否第一次就加载渲染日历
        sourceFn: null, // 数据源
        tipText: "yyyy-mm-dd", //文本日期格式
        frequent: false, // 单月显示
        titleTip: "{{year}}年{{month}}月", // 日历标题格式
        todayInfos: ["", "", ""],
        rangeColor: "#F0F5FB",  // 区间
        hoverColor: "#d9e5f4",  // 鼠标滑过
        control: true, // 控制翻页按钮是否显示
        showPrev: false, // 控制上个月翻月按钮
        showNext: true, // 控制下个月翻月按钮
        startDelayDays: 0, // 在开始的基础上叠加天数 配合fatalism一起使用
        fatalism: 0, // 天数
        mos: 6, //月份
        isTodayClick: false, // 当天是否可点击
        weeks: ['星期日', '星期一', '星期二', '星期三', '星期四', '星期五', '星期六'],
        classNames: {
            week: ["sun", "mon", "tue", "wed", "thu", "fri", "sat"],
            caldate: "caldate",
            nodate: "nodate", // 禁用和空
            today: "today", // 今天
            hover: "hover", // 鼠标滑过效果
            selectDay: "selectDay", // input 
            interval: "interval", // 区间
            festival: "calfest", // 节日

            monthNext: "month-next"
        },
        // 级联
        cascade: {
            days: 1, // 天数叠加一天
            trigger: ".calendar",
            isTodayClick: false
        },
       // 日历模板
